<script>
(function(){
  const root = document.currentScript.closest('.parlios-hub');
  if(!root) return;
  // Tabs
  root.querySelectorAll('.side-link').forEach(btn=>{
    btn.addEventListener('click', ()=>{
      root.querySelectorAll('.side-link').forEach(b=>b.classList.remove('is-active'));
      btn.classList.add('is-active');
      const tab = btn.getAttribute('data-tab');
      root.querySelectorAll('.tab').forEach(p=>p.classList.toggle('is-active', p.getAttribute('data-tab')===tab));
    });
  });
  // Fake filters
  root.querySelectorAll('.chip').forEach(ch=>{
    ch.addEventListener('click', ()=>{
      root.querySelectorAll('.chip').forEach(c=>c.classList.remove('is-active'));
      ch.classList.add('is-active');
    });
  });
})();
</script>
