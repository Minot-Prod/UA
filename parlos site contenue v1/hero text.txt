<script>
/* Minimal i18n switcher (scoped to this section) */
(function(){
  const root = document.currentScript.closest('.hero-parlios');
  if(!root) return;
  const t = (lang) => {
    root.querySelectorAll('[data-i18n]').forEach(el=>{
      const key = lang === 'en' ? 'data-en' : 'data-fr';
      const val = el.getAttribute(key); if(val) el.textContent = val;
    });
    root.querySelectorAll('.lang-btn').forEach(b=>{
      const is = b.getAttribute('data-lang')===lang;
      b.classList.toggle('is-active', is);
      b.setAttribute('aria-pressed', is);
    });
  };
  root.querySelectorAll('.lang-btn').forEach(btn=>{
    btn.addEventListener('click', ()=> t(btn.getAttribute('data-lang')));
  });
})();
</script>
